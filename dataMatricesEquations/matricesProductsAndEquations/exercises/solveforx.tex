\begin{exercise}


Solve for $x, y, z, w$:
\[
\left(\begin{array}{ccc}
3x & 8 & 6\\
2z & 9 & 0
\end{array}\right) + 2 \left(\begin{array}{ccc} 
5& -4 & 4y\\
3 & 7 & 6
\end{array}\right) = \left(\begin{array}{ccc}
2y & 0 & z\\
10 & w & 12 
\end{array}\right)
\]
\begin{prompt}
\[
x = \answer{-11/3}, \hspace{.4cm} y = \answer{-1/2}, \hspace{.4cm} z = \answer{2}, \hspace{.4cm} w = \answer{23}
\]
\end{prompt}



















\end{exercise}
