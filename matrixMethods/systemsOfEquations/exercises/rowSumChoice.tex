\begin{exercise}
  Suppose I have a matrix:
  \[
  \begin{pmatrix}
    -3 & 1 & 0 & 2 \\
    3 & -1 & -2 & 1 \\
    0 & 3 & -3 & -1
  \end{pmatrix}
  \]
  The row operation $R_2+3R_1$ can be plugged into\dots
  \begin{selectAll}
    \choice[correct]{Row one, $R_1$.}
    \choice[correct]{Row two, $R_2$.}
    \choice{Row three, $R_3$.}
  \end{selectAll}
\end{exercise}
