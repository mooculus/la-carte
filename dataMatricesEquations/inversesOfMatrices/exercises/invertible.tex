\begin{exercise}
Which of the following matrices are invertible?

\begin{hint}
Two rows being a multiple of each other, produces a row of zeros in the first half of the matrix $(A \hspace{.08cm}|\hspace{.08cm} I)$.
\end{hint}

\begin{multipleChoice}
\choice[correct]{ \[\left(\begin{array}{cccc}
5 & 7 & -10 & 1\\
0 & 2 & -8 & -7\\
0 & 0 & 0 & 4\\
0 & 0 & 0 & -9\\
\end{array}\right)
\]}

\choice{\[\left(\begin{array}{cccc}
3 & 0 & -1 & 4\\
2 & -1 & 0 & 5\\
0 & 3& 5 & -8\\
4 & -2 & 0 & 10\\
\end{array}\right)
\]}

\choice{\[ \left(\begin{array}{cc}
2 & 0 \\
0 & 3\\
8 & 9
\end{array}\right)\]}

\choice[correct]{\[\left(\begin{array}{cccc}
0& -2 & -5 & 7\\
-2& 0 & -9 & -1\\
-5& -9 & 0 & 8\\
7& -1 & 8 & 0
\end{array}\right)\]}

\choice{\[\left(\begin{array}{ccccc}
1& 0 & 0 & 0&0\\
0& 3 & 0 & 0&0\\
0& 0 & 7 & 1&0\\
0& 9 & 0 & 0&0\\
0& 0 & 0 & 0&5
\end{array}\right)\]}
\end{multipleChoice}
\end{exercise}
