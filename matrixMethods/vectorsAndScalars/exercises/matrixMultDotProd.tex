\begin{exercise}
  In this problem we are working with column vectors and they all have
  the same dimension. Given column vectors $\vec{v}$ and $\vec{w}$ we
  define the \dfn{dot product} via:
  \[
  \vec{v}\dotp \vec{w} = \vec{v}^\transpose \vec{w}
  \]
  Armed with this fact, which of the following expressions represents
  a matrix multiplied by a vector?
  \begin{multipleChoice}
    \choice{$\begin{pmatrix} \vec v_1  &  \vec v_2  &  \vec v_3 \end{pmatrix} \vec{w} = \begin{pmatrix}\vec{v}_1\dotp \vec w\\ \vec{v}_2\dotp \vec{w}\\ \vec{v}_3\dotp \vec w\end{pmatrix}$}
    \choice[correct]{$\begin{pmatrix} \vec v_1^\transpose \\ \vec v_2^\transpose \\ \vec v_3^\transpose\end{pmatrix} \vec{w} = \begin{pmatrix}\vec{v}_1\dotp \vec w\\ \vec{v}_2\dotp \vec{w}\\ \vec{v}_3\dotp \vec w\end{pmatrix}$}
    \choice{$\begin{pmatrix} \vec v_1^\transpose  &  \vec v_2^\transpose  &  \vec v_3^\transpose \end{pmatrix} \vec{w} = \begin{pmatrix}\vec{v}_1\dotp \vec w\\ \vec{v}_2\dotp \vec{w}\\ \vec{v}_3\dotp \vec w\end{pmatrix}$}
    \choice{$\begin{pmatrix} \vec v_1  \\ \vec v_2  \\ \vec v_3 \end{pmatrix} \vec{w} = \begin{pmatrix}\vec{v}_1\dotp \vec w\\ \vec{v}_2\dotp \vec{w}\\ \vec{v}_3\dotp \vec w\end{pmatrix}$}
  \end{multipleChoice}
  \begin{feedback}[correct]
    We see that in some sense, matrix multiplication can be viewed as
    the dot product distributing over the matrix.
  \end{feedback}
\end{exercise}
