\begin{exercise}

  Consider the following matrices:
  \[
    A =
    \begin{pmatrix}
      0 & 7\\
      -2 & 0\\
      0 & -5 \\
      0 & 0
    \end{pmatrix}, \quad
    B =
    \begin{pmatrix}
      8 & -1 & 0 & -3\\
      2 & -4 & 5 & -2
    \end{pmatrix}
  \]

  % Are both multiplications $AB$ and $BA$ possible? If not write
  % ``NA'' for the entries of the one that is not.
  Find $AB$ and $BA$.

  \begin{prompt}
    \begin{align*}
      AB
      & =
        \begin{pmatrix}
          \answer{14} & \answer{-28} & \answer{35} & \answer{-14}\\
          \answer{-16} & \answer{2} & \answer{0} & \answer{6}\\
          \answer{-10} & \answer{20} & \answer{-25} & \answer{10}\\
          \answer{0} & \answer{0} & \answer{0} & \answer{0}
        \end{pmatrix} \\
      BA
      & =
        \begin{pmatrix}
          \answer{2} & \answer{56}\\
          \answer{8} & \answer{-11}
        \end{pmatrix}
    \end{align*}
  \end{prompt}

\end{exercise}
