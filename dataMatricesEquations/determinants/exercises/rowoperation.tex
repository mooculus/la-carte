\begin{exercise}


Find the determinant using row operations to convert to a triangular matrix.
\[
A = \begin{pmatrix}
2 & 0 & 4\\
-1 & 0 & -3\\
1 & 1 & 2
\end{pmatrix}
\]

\begin{prompt}
The triangular matrix is:
\[\begin{pmatrix}
1 & \answer{1} & \answer{2}\\
\answer{0} & 1 & \answer{-1}\\
\answer{0} & \answer{0} & -2
\end{pmatrix}\]
giving,
\[det(A) = \answer{2}\]
\end{prompt}


































\end{exercise}
