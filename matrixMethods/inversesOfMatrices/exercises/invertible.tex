\begin{exercise}
Which of the following matrices are invertible?
\begin{multipleChoice}
\choice[correct]{ \[\left(\begin{array}{cccc}
5 & 7 & -10 & 1\\
0 & 2 & -8 & -7\\
0 & 0 & 0 & 4\\
0 & 0 & 0 & -9\\
\end{array}\right)
\]}

\choice{\[\left(\begin{array}{cccc}
3 & 0 & -1 & 4\\
2 & -1 & 0 & 5\\
0 & 3& 5 & -8\\
4 & -2 & 0 & 11\\
\end{array}\right)
\]
Note that $R_{4}= 2R_{2}$.}

\choice{\[ \left(\begin{array}{cc}
2 & 0 \\
0 & 3\\
8 & 9 
\end{array}\right)\]}

\choice[correct]{\[\left(\begin{array}{cccc}

-2& 0 & -9 & -1\\

\end{array}\right)\]}

\choice{\[\left(\begin{array}{ccccc}

\end{array}\right)\]}
\end{multipleChoice}
\end{exercise}
