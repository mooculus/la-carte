\begin{exercise}
Compute the determinant of the following matrices:
\[
A=
\begin{pmatrix}
-1 & 5\\
4 & -7
\end{pmatrix},\  B = \begin{pmatrix} 0 & 6\\
-1 & 0
\end{pmatrix},\ C= \begin{pmatrix} 2 & -3 \\
0 & 0
\end{pmatrix},\ D = \begin{pmatrix} 2 & -8 \\
1 & -4
\end{pmatrix}
\]
\begin{prompt}
\begin{align*}
\det(A) &= \answer{-13}\\
\det(B) &= \answer{6}\\ 
\det(C) &= \answer{0}\\
\det(D) &= \answer{0}
\end{align*}
\end{prompt}
\end{exercise}
