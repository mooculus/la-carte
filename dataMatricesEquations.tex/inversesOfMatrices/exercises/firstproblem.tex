\begin{exercise}
Find the inverse of the following matrices.
\[
A= \begin{pmatrix}
  7 & 10   \\
  -1 &  0
\end{pmatrix}, \quad
B= \begin{pmatrix}
  0 & 5   \\
  -2 &  0
\end{pmatrix}, \quad
C= \begin{pmatrix}
  -4 & -8   \\
  -3 &  -1
\end{pmatrix}
\]
\begin{prompt}
\[A^{-1}=  \begin{pmatrix}
  \answer{0} & \answer{-1}   \\
  \answer{1/10} & \answer{7/10}  
\end{pmatrix},\quad  B^{-1}=  \begin{pmatrix}
  \answer{0} & \answer{-1/2}   \\
  \answer{1/5} & \answer{0}  
\end{pmatrix},
\]
\[ C^{-1}=  \begin{pmatrix}
  \answer{1/20} & \answer{-2/5}   \\
  \answer{-3/20} & \answer{1/5}  
\end{pmatrix}
\]
\end{prompt}
\end{exercise}
